Verde (Manipulação)

push() 
let arr = [1,2];
arr.push(3);
console.log(arr); // [1,2,3]
pop() 
let arr = [1,2,3];
arr.pop();
console.log(arr); // [1,2]
shift() 
let arr = [1,2,3];
arr.shift();
console.log(arr); // [2,3]
unshift() 
let arr = [2,3];
arr.unshift(1);
console.log(arr); // [1,2,3]

Amarelo (Busca)

find() 
let arr = [1,2,3];
let x = arr.find(n => n > 1);
console.log(x); // 2
some() 
let arr = [1,2,3];
let tem = arr.some(n => n > 2);
console.log(tem); // true

Vermelho (Verificação)

every() 
let arr = [2,4,6];
let todos = arr.every(n => n % 2 === 0);
console.log(todos); // true
