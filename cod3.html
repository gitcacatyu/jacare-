Verde (Busca)

indexOf() 
let arr = [1,2,3,2];
console.log(arr.indexOf(2)); // 1
lastIndexOf() 
let arr = [1,2,3,2];
console.log(arr.lastIndexOf(2)); // 3

Amarelo (Manipulação)

reverse() 
let arr = [1,2,3];
arr.reverse();
console.log(arr); // [3,2,1]
concat() 
let a = [1,2];
let b = [3,4];
console.log(a.concat(b)); // [1,2,3,4]
join() 
let arr = [1,2,3];
console.log(arr.join('-')); // "1-2-3"

Vermelho (Conversão)

toString() 
let arr = [1,2,3];
console.log(arr.toString()); // "1,2,3"
