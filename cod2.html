Verde (Manipulação)

sort() 
let arr = [3,1,2];
arr.sort();
console.log(arr); // [1,2,3]
includes() 
let arr = [1,2,3];
console.log(arr.includes(2)); // true
slice() 
let arr = [1,2,3,4];
console.log(arr.slice(1,3)); // [2,3]

Amarelo (Transformação)

map() 
let arr = [1,2,3];
let novo = arr.map(n => n*2);
console.log(novo); // [2,4,6]
filter() 
let arr = [1,2,3,4];
let pares = arr.filter(n => n%2===0);
console.log(pares); // [2,4]

Vermelho (Redução)

reduce() 
let arr = [1,2,3];
let soma = arr.reduce((acc,n) => acc+n, 0);
console.log(soma); // 6
forEach() 
let arr = [1,2,3];
arr.forEach(n => console.log(n));
// 1
// 2
// 3
